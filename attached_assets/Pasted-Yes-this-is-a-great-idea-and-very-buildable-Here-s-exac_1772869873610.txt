```python
async def get_library_context() -> str:
    """Build reference context from DB for AI prompt"""
    items = await db.fetch_all(
        "SELECT item_name, cubic_yards, is_special, special_fee 
         FROM item_reference_library 
         ORDER BY times_seen DESC LIMIT 150"
    )
    if not items:
        return ""
    
    lines = ["KNOWN ITEM REFERENCE (use these CY values):"]
    for item in items:
        line = f"- {item['item_name']}: {item['cubic_yards']} CY"
        if item['is_special']:
            line += f" [SPECIAL FEE: ${item['special_fee']}]"
        lines.append(line)
    return "\n".join(lines)
```

Inject into Pass 1 prompt:
{library_context}
Now analyze these photos and identify all items.

### Pass 3: Web Lookup for Unknown Items
After Pass 2, for each item in items_needing_lookup:
```python
async def lookup_item_dimensions(item_name: str) -> dict:
    """Search web for item dimensions and calculate CY"""
    try:
        async with httpx.AsyncClient() as client:
            # Search Tavily for dimensions
            response = await client.post(
                "https://api.tavily.com/search",
                json={
                    "api_key": os.getenv("TAVILY_API_KEY"),
                    "query": f"{item_name} dimensions inches length width height size",
                    "search_depth": "basic",
                    "max_results": 3
                },
                timeout=8.0
            )
            data = response.json()
            content = " ".join([r.get("content", "") 
                               for r in data.get("results", [])])
            
            # Ask Claude to extract dimensions and calculate CY
            calc_response = await anthropic_client.messages.create(
                model="claude-sonnet-4-20250514",
                max_tokens=200,
                messages=[{
                    "role": "user",
                    "content": f"""From this text, extract the dimensions 
                    of a {item_name} and calculate cubic yards.
                    
                    Text: {content[:2000]}
                    
                    Return ONLY JSON:
                    {{
                        "length_inches": 0,
                        "width_inches": 0, 
                        "height_inches": 0,
                        "cubic_yards": 0.0,
                        "confidence": 0.8,
                        "source_note": "brief note on what was found"
                    }}
                    
                    Cubic yards = (L x W x H in inches) / 46656
                    Round cubic_yards to 2 decimal places.
                    If you cannot find dimensions, return cubic_yards: 0"""
                }]
            )
            
            result = json.loads(calc_response.content[0].text)
            
            if result.get("cubic_yards", 0) > 0:
                # Save to reference library
                await db.execute("""
                    INSERT INTO item_reference_library 
                    (item_name, cubic_yards, confidence, source, 
                     search_query_used, times_seen)
                    VALUES (:name, :cy, :conf, 'web_search', 
                            :query, 1)
                    ON CONFLICT (item_name) DO UPDATE SET
                    times_seen = item_reference_library.times_seen + 1,
                    updated_at = NOW()
                """, {
                    "name": item_name.lower(),
                    "cy": result["cubic_yards"],
                    "conf": result["confidence"],
                    "query": f"{item_name} dimensions"
                })
                
                return result
                
    except Exception as e:
        print(f"Lookup failed for {item_name}: {e}")
    
    return {"cubic_yards": 0, "confidence": 0}
```

```python
async def update_library_from_estimate(items: list):
    for item in items:
        normalized_name = item['name'].lower().strip()
        
        # Check if item exists in library
        existing = await db.fetch_one(
            "SELECT id, cubic_yards FROM item_reference_library 
             WHERE item_name = :name",
            {"name": normalized_name}
        )
        
        if existing:
            # Increment times_seen
            await db.execute(
                "UPDATE item_reference_library 
                 SET times_seen = times_seen + 1,
                 updated_at = NOW()
                 WHERE item_name = :name",
                {"name": normalized_name}
            )
        else:
            # New item found by AI — add to library
            if item.get('cubic_yards', 0) > 0:
                await db.execute("""
                    INSERT INTO item_reference_library
                    (item_name, cubic_yards, confidence, 
                     source, times_seen)
                    VALUES (:name, :cy, 0.7, 'ai_learned', 1)
                    ON CONFLICT (item_name) DO NOTHING
                """, {
                    "name": normalized_name,
                    "cy": item['cubic_yards']
                })
```
